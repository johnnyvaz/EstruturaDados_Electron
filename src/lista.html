            function List(capacity) {
                var collection = [];
    
                if (capacity != null) {
                    collection[capacity];
                }
                ///busca um elemento
                this.GetItem = function (index) {
                    if (index <= collection.length - 1) {
                        return collection[index];
                    }
                    else {
                        alert("Não encontrado");
                    }
                }
                ///insere um elemento em uma posição
                this.SetItem = function (index, item) {
                    if (index <= collection.length - 1) {
                        collection[index] = item;
                    }
                    else {
                        alert("Posição não existente! verifique.");
                    }
                }
                ///Adiciona um elemento no fin da lista
                this.Add = function (item) {
                    collection[collection.length] = item;
                }
                ///Adiciona os elementos da coleção especificada ao final das coleções.
                this.AddRange = function (items) {
                    for (var i = 0; i < items.Count(); i++) {
                        collection[collection.length] = items.GetItem(i);
                    }
                }
                ///Insere um elemento no System.Collections.Generic.List  no índice especificado.
                    this.Insert = function (pos, item) {
                        if (pos <= collection.length - 1) {
                            collection.splice(pos, 0, item);
                        }
                        else {
                            throw "Index must be within the bounds of the List";
                        }
                    }   
                    ///limpa a lista
                    this.Clear = function () {
                        collection = [];
                    }
                    ///tamanho da lista.
                    this.Count = function () {
                        return collection.length;
                    }
                    ///comparação de elementos
                    this.Equal = function (x, y) {
                        return (x === y);
                    }
                    ///busca elementos
                    this.Find = function (obj) {
                        var index = -1;
                        
                        for (var i = 0; i < collection.length; i++) {
                            var item = collection[i];
                            
                            if (this.Equal(item, obj)) {
                                index = i;
                                break;
                            }
                        }   
                        return index;
                    }
                    ///verifica se existe
                    this.Exists = function (obj) {
                        var isFind = false;
                        
                        if (this.Find(obj) > -1) {
                            isFind = true;
                        }
                        
                        return isFind;
                    }
                    ///Executa a ação especificada em cada elemento das listas
                    this.ForEach = function (action) {
                        for (var i = 0; i < collection.length; i++) {
                            var item = collection[i];
                            
                            action(item);
                        }
                    }
                    ///Remove o elemento no índice especificado
                    this.RemoveAt = function (index) {
                        collection.splice(index, 1);
                    }
                    ///remove vários elementos
                    this.RemoveRange = function (index, count) {
                        collection.splice(index, count);
                    }
                    ///
                    this.GetRange = function (index, count) {
                        newList = new List();
                        
                        var indexCount = 0;
                        
                        for (var i = index; i < collection.length; i++) {
                            var item = collection[i];
                            
                            if (indexCount < count) {
                                newList.Add(item);
                            }
                            else {
                                break;
                            }
                            
                            indexCount++;
                        }
                        
                        return newList;
                    }
                }
